In previous lessons, we've used the random function to generate random numbers.

Given an array as an argument, random can also be used to choose a random element from an array.

// choose a random fruit function setup() { var fruit = ["banana", "apple", "orange"]; console.log(random(fruit)); }

In the following example we can use random with an array of sounds to play random sounds.

var sounds = []; var r, g, b; function preload() { sounds.push(loadSound('ex0.wav')); sounds.push(loadSound('ex1.wav')); sounds.push(loadSound('ex2.wav')); sounds.push(loadSound('ex3.wav')); } function draw(){ background(0); fill(r, g, b); ellipse(width/2, height/2, 200, 200); } function mousePressed() { var d = dist(mouseX, mouseY, width/2, height/2); if (d < 100) { r = random(0, 255); g = random(0, 255); b = random(0, 255); random(sounds).play(); } }
