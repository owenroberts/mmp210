p5: drawing with loops

MMP 210 > Week 4

In p5, we can use loop to draw many more shapes and colors with much less code. Let's look at a few of our old examples, reworked using loops.

Week 2: Changing color:

function setup() {
    createCanvas(640, 200);
    background("pink");
    noStroke();

    var y = 100;
    var g = 0;
    var b = 0;
    var rIncrement = 50;

    for (var r = 0; r < width; r += rIncrement) {
        fill(r, g, b);
        ellipse(r, y, rIncrement, rIncrement);
    }
}

Week 3: Grid with static variables:

function setup() {
    createCanvas(640, 480);
    background("white");
    
    stroke("darkblue");
    strokeWeight(2);

    var x = 0;
    var y = 0;
    
    var ratio = 10;
    var xunit = width/ratio;
    var yunit = height/ratio;

    for (var i = 0; i < ratio - 1; i++) {
        x += xunit;
        y += yunit;
        line(x, 0, x, height);
        line(0, y, width, y);
    }
}

For one more example, let's add in a conditional statment to our ellipse example.

function setup() {
    createCanvas(640, 480);
    background("white");
    
    stroke("darkblue");
    strokeWeight(1);

    var x = 0;
    var y = 0;
    var radius = 40;
    
    var ratio = 4;
    var xunit = width/ratio;
    var yunit = height/ratio;

    for (var i = 0; i < ratio * ratio; i++) {
        if ( x >= width) {
            x = 0;
            y += yunit;
        }        
        if (x >= width/2) fill("lightblue");
        else fill("pink");
        ellipse(x + xunit/2, y + yunit/2, radius, radius);
        x += xunit;
    }
}
