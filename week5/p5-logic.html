function setup() {
	createCanvas(640, 480);
}

function draw() {
	strokeWeight(4);
	stroke("blue");

	if (mouseX > width/2) {
		fill("lightblue");
	} else {

	}

Quadrants

In the next example we will divide the canvas into quadrants:

/* initialize red green and blue all to zero */
var r = g = b = 0;
function setup() {
	createCanvas(640, 360);
}

function draw() {
	background(r, g, b);
	
	/* draw lines for quadrants */
	strokeWeight(1);
	stroke("white");
	line(width/2, 0, width/2, height);
	line(0, height/2, width, height/2);

	if (mouseX > width/2) {
		r += 1;
	} else {
		r -= 1;
	}
	
	if (mouseX < width/2) {
		g += 1;
	} else {
		g -= 1;
	}
	
	if (mouseY > height/2) {
		b += 1;
	} else {
		b -= 1;
	}
}
