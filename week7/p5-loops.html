var column = 40;
var row = 20;
function setup() { 
  createCanvas(640, 360);
	rectMode(CENTER);
	for (var x = 0; x <= width; x += column) {
		line(x, 0, x, height);
		for (var y = 0; y <= height; y += row) {
			line(0, y, width, y);
		}
	}
}

function draw() { 
	var sz = 10;
	for (var x = 0; x < width; x += sz) {
			//line(x, 0, x, height);
		if (mouseX > x) {
			fill(mouseX-x);
		} else {
			fill(x-mouseX);
		}
		rect(x,0,sz, height);
	}
}
